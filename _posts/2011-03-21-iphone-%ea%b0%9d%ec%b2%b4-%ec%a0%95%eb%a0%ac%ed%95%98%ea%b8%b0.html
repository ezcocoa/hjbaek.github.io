NSSortDescriptor *category1Descriptor = [[NSSortDescriptor alloc] initWithKey:@"category1" ascending:YES];
NSSortDescriptor *category2Descriptor = [[NSSortDescriptor alloc] initWithKey:@"category2" ascending:YES];
