iOS5 업데이트 된 이후에 일본어 키보드 위에 한자 표시 칸이 추가 되었습니다.
이로 인해 일본어 키보드가 불려졌을 때 사이즈 대응을 해야 합니다.
아래는 키보드 Notification 후킹하기 위한 Notification 등록입니다.
[sourcecode language="ObjC"]
[[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(keyboardWillShow:) name:UIKeyboardWillShowNotification object:nil];
[/sourcecode]
키보드의 사이즈를 구하는 메서드입니다.
[sourcecode language="ObjC"]
NSDictionary *info = [notification userInfo];
CGSize kbSize = [[info objectForKey:UIKeyboardFrameEndUserInfoKey] CGRectValue].size;
NSLog(@"%@, %f, %f", info, kbSize.width, kbSize.height);
_textView.text = [NSString stringWithFormat:@"%@",info];
[/sourcecode]
