+ (UIColor *)colorWithHex:(NSString *)hexString Alpha:(CGFloat)alpha{
NSUInteger colorRGB[3];
for(NSInteger i=0; i<3; i++) {
[[NSScanner scannerWithString:[hexString substringWithRange:NSMakeRange(i * 2, 2)]] scanHexInt:colorRGB + i];
}
return [UIColor colorWithRed:(CGFloat)(colorRGB[0]/255.0f) green:(CGFloat)(colorRGB[1]/255.0f) blue:(CGFloat)(colorRGB[2]/255.0f) alpha:alpha];
}
self.window.backgroundColor = [UIColor colorWithHex:@"ff2356" Alpha:1.0];
