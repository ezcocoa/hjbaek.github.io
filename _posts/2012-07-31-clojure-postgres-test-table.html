CREATE TABLE category (
id integer PRIMARY KEY,
name text
);
CREATE TABLE tag (
id integer PRIMARY KEY,
name text
);
CREATE TABLE post (
id integer PRIMARY KEY,
title text,
content text,
category_id integer REFERENCES category (id)
);
CREATE TABLE comment (
id integer PRIMARY KEY,
post_id integer REFERENCES post (id),
content text
);
CREATE TABLE post_tag (
id integer PRIMARY KEY,
post_id integer REFERENCES post (id),
tag_id integer REFERENCES tag (id)
);
[/code]
