1. 상단 타이틀 없애기.
[sourcecode language="xml"]
android:theme="@android:style/Theme.NoTitleBar"
[/sourcecode]
2. 화면에 따라 상단 타이틀 바꾸기.
[sourcecode lanugage="java"]
setTitle(getString(R.string.app_name));
[/sourcecode]
